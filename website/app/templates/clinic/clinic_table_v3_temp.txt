    <script>
       document.addEventListener('DOMContentLoaded', function() {
        const searchInput = document.getElementById('search');
    const searchButton = document.getElementById('searchButton');

    // Функция для выполнения поиска
    function performSearch() {
        const query = searchInput.value.trim();
        if (query.length >= 3) {
            fetch(`/clinic/clinic_data?name=${encodeURIComponent(query)}`)
                .then(response => response.json())
                .then(data => {
                    updateTable(data);
                })
                .catch(error => console.error('Error fetching patients:', error));
        } else {
            fetch('/clinic/clinic_data')
                .then(response => response.json())
                .then(data => {
                    updateTable(data);
                });
        }
    }

    // Обработчик нажатия кнопки поиска
    searchButton.addEventListener('click', performSearch);

    // Обработчик нажатия клавиши Enter
    searchInput.addEventListener('keypress', function(event) {
        if (event.key === 'Enter') {
            performSearch();
        }
    });

    function formatDateTime(dateTimeString) {
        if (!dateTimeString) return '';
        
        const date = new Date(dateTimeString);
        const day = String(date.getDate()).padStart(2, '0');
        const month = String(date.getMonth() + 1).padStart(2, '0'); // Months are 0-based
        const year = date.getFullYear();
        const hours = String(date.getHours()).padStart(2, '0');
        const minutes = String(date.getMinutes()).padStart(2, '0');
        
        return `${day}.${month}.${year} в ${hours}:${minutes}`;
    }


    fetch('/clinic/clinic_data')
        .then(response => response.json())
        .then(data => {
            updateTable(data);
        });

    function updateTable(patients) {
        const tableBody = document.querySelector('#clinicTable tbody');
        tableBody.innerHTML = '';

        patients.forEach(patient => {
            const row = document.createElement('tr');
            row.innerHTML = `
                <td>${patient.id != null ? patient.id : ''}</td>
                <td>${patient.current_time != null ? formatDateTime(patient.current_time) : ''}</td>
                <td>${patient.card_number != null ? patient.card_number : ''}</td>
                <td>${patient.full_name != null ? patient.full_name : ''}</td>
                <td>${patient.birthday != null ? patient.birthday : ''}</td>
                <td>${patient.directed != null ? patient.directed : ''}</td>
                <td>${patient.military_unit != null ? patient.military_unit : ''}</td>
                <td>${patient.military_rank != null ? patient.military_rank : ''}</td>
                <td>${patient.belonging != null ? patient.belonging : ''}</td>
                <td>${patient.address != null ? patient.address : ''}</td>
                <td>${patient.hostilities != null ? (patient.hostilities ? 'Да' : 'Нет') : ''}</td>
                <td>${patient.military_district != null ? patient.military_district : ''}</td>
                <td>${patient.phone_number != null ? patient.phone_number : ''}</td>
                <td>${patient.patient_info != null ? patient.patient_info : ''}</td>
                <td><button class="btn btn-primary info-btn" data-id="${patient.id}">Информация</button></td>
                <td><button class="btn btn-danger delete-btn" data-id="${patient.id}">Удалить</button></td>
            `;
            tableBody.appendChild(row);
        });

        // Add event listener to info buttons
        document.querySelectorAll('.info-btn').forEach(button => {
            button.addEventListener('click', function() {
                const patientId = this.getAttribute('data-id');
                window.location.href = `/clinic/patient/${patientId}`;
            });
        });

        document.getElementById('clinicDataForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData.entries());
            data.hostilities = document.getElementById('editHostilities').value === 'true';
            fetch('/clinic/clinic_data/', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(data)
            })
            .then(response => response.json())
            .then(data => {
                if (data.detail) {
                    alert("Ошибка: " + data.detail);
                } else {
                    fetch('/clinic/clinic_data')
                        .then(response => response.json())
                        .then(data => {
                            updateTable(data);
                            $('#addPatientModal').modal('hide'); // Закрытие модального окна после успешного добавления пациента
                    });
                    $('#addPatientModal').modal('hide');
                }
            })
            .catch(error => alert("Ошибка: " + error.message));
        });

        document.getElementById('totalPatients').textContent = `${patients.length} пациентов`;
    }
});
